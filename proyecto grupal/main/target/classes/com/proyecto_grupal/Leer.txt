create database proyectogrupal

//creo tabla sueldo

CREATE TABLE proyectogrupal.sueldo (
	id_sueldo INT auto_increment NOT NULL,
	sueldoBruto INT NULL,
	sueldoNeto INT NULL,
	CONSTRAINT sueldo_PK PRIMARY KEY (id_sueldo))

//creo la tabla empleado

CREATE TABLE proyectogrupal.empleado (
	id_empleado INT auto_increment NOT NULL,
	nombre varchar(30) NULL,
	apellido varchar(30) NULL,
	fecha_nacimiento DATE NULL,
	id_sueldo INT NULL,
	CONSTRAINT empleado_PK PRIMARY KEY (id_empleado),
	CONSTRAINT empleado_FK FOREIGN KEY (id_sueldo) REFERENCES proyectogrupal.sueldo(id_sueldo)
)

//creo la tabla valor_hora

CREATE TABLE proyectogrupal.valor_hora (
	id_hora INT auto_increment NOT NULL,
	valorHora INT NULL,
	valorHoraExtra INT NULL,
	CONSTRAINT valor_hora_PK PRIMARY KEY (id_hora)
)

//creo la tabla jornada laboral

CREATE TABLE proyectogrupal.jornadaLaboral (
	id_jornada INT auto_increment NOT NULL,
	diaLaboral ENUM ('lunes','martes','miercoles','jueves','viernes') NULL,
	horario_entrada TIME NULL,
	horario_salida TIME NULL,
	id_empleado INT NULL,
	id_hora INT NULL,
	CONSTRAINT jornadaLaboral_PK PRIMARY KEY (id_jornada),
	CONSTRAINT jornadaLaboral_FK FOREIGN KEY (id_empleado) REFERENCES proyectogrupal.empleado(id_empleado),
	CONSTRAINT jornadaLaboral_FK_1 FOREIGN KEY (id_hora) REFERENCES proyectogrupal.valor_hora(id_hora)
)
